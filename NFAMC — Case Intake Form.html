Server-side sample (Node.js + Express)

Below is a sample Express endpoint (using multer) that accepts form submissions and file uploads. Save this as server_sample.js, run npm install express multer, then node server_sample.js. This is only an example â€” adapt to your authentication, storage, and validation requirements.

const express = require('express');
const multer = require('multer');
const path = require('path');
const app = express();

// Simple local disk storage (for demo only). In production, use secure storage.
const storage = multer.diskStorage({
  destination: (req, file, cb) => cb(null, path.join(__dirname, 'uploads')),
  filename: (req, file, cb) => cb(null, Date.now() + '-' + file.originalname)
});
const upload = multer({ storage });

app.use(express.json());
app.use(express.urlencoded({ extended: true }));

app.post('/submit-case', upload.array('attachments'), (req, res) => {
  try {
    const body = req.body; // all form fields
    const files = req.files.map(f => ({ filename: f.filename, original: f.originalname, size: f.size }));
    // TODO: validate, verify signature, save to DB, notify assigned officer, etc.
    console.log('Received case', body.caseNumber || '(no case number)', 'files:', files);
    return res.json({ ok: true, received: body, files });
  } catch (err) {
    console.error('Upload error', err);
    return res.status(500).json({ ok: false, error: 'Server error' });
  }
});

app.listen(3000, () => console.log('Server running on http://localhost:3000'));
